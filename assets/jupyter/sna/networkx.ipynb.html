# Some examples of bipartite, cycle, and star graph
A = nx.complete_bipartite_graph(3,4)
B = nx.cycle_graph(4)
C = nx.star_graph(6)

plt.subplot(221)
plt.title("Bipartite")
top = nx.bipartite.sets(A)[0]
nx.draw(A, pos=nx.bipartite_layout(A, top))
plt.subplot(222)
plt.title("Cycle")
nx.draw(B, pos=nx.circular_layout(B))
plt.subplot(223)
plt.title("Star")
nx.draw(C)
